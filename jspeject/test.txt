// Function to read EXIF data using piexifjs
function readExifData(image) {
  const binaryData = image.src.split(",")[1]; // Get the image data from the base64 string
  const exifData = piexif.load(binaryData);
  
  // Extract specific EXIF data tags
  const make = exifData["0th"][piexif.ImageIFD.Make] || "Unknown";
  const model = exifData["0th"][piexif.ImageIFD.Model] || "Unknown";
  const dateTime = exifData["0th"][piexif.ImageIFD.DateTime] || "Unknown";
  
  return { make, model, dateTime };
}


















document.getElementById("metadata-form").addEventListener("submit", function(e) {
  e.preventDefault();
  
  // Get new EXIF values from the form
  const make = document.getElementById("make").value;
  const model = document.getElementById("model").value;
  const dateTime = document.getElementById("dateTime").value;
  
  const file = document.getElementById("file-input").files[0];
  const reader = new FileReader();
  
  reader.onload = function(e) {
    const image = new Image();
    image.onload = function() {
      // Read existing EXIF data from the image
      const binaryData = image.src.split(",")[1];
      const exifData = piexif.load(binaryData);
      
      // Modify the EXIF data
      exifData["0th"][piexif.ImageIFD.Make] = make;
      exifData["0th"][piexif.ImageIFD.Model] = model;
      exifData["0th"][piexif.ImageIFD.DateTime] = dateTime;
      
      // Convert modified EXIF data back to bytes
      const newExifBytes = piexif.dump(exifData);
      
      // Insert modified EXIF data back into the image
      const newImageData = piexif.insert(newExifBytes, image.src);
      
      // Create a new image with updated EXIF data
      const canvas = document.createElement("canvas");
      const ctx = canvas.getContext("2d");
      canvas.width = image.width;
      canvas.height = image.height;
      ctx.drawImage(image, 0, 0);
      
      // Convert the canvas image back to a JPEG with updated EXIF data
      canvas.toBlob(function(blob) {
        const a = document.createElement("a");
        a.href = URL.createObjectURL(blob);
        a.download = "updated_image.jpg";  // Download the updated image
        a.click();  // Trigger the download
        alert("Image updated and ready to download.");
      }, "image/jpeg");
    };
    image.src = e.target.result;
  };
  
  reader.readAsDataURL(file);  // Read the file as a data URL (base64)
});
